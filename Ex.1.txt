import 'dart:io';

void main() {
  print('Digite um ponto:');
  int x1 = int.parse(stdin.readLineSync()!);

  print('Digite um ponto:');
  int x2 = int.parse(stdin.readLineSync()!);

  print('Digite um ponto:');
  int y1 = int.parse(stdin.readLineSync()!);

  print('Digite um ponto:');
  int y2 = int.parse(stdin.readLineSync()!);

  double D = sqrt((x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1));

  print(D);
}
